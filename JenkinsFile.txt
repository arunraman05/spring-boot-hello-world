pipeline {
    agent any

    tools {
        // Use the Maven tool configured in Jenkins (must match the name in Jenkins global tools)
        maven 'M3'
    }

    environment {
        DOCKER_IMAGE   = 'spring-boot-hello-world'
        CONTAINER_NAME = 'spring-boot-hello-world-container'
    }

    parameters {
        choice(
            name: 'ENVIRONMENT',
            choices: ['testing', 'prod'],
            description: 'Select environment to run pipeline'
        )
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/arunraman05/spring-boot-hello-world.git'
            }
        }

        stage('Print Path') {
            steps {
                bat 'echo %PATH%'
            }
        }

        stage('Maven Test') {
            when {
                expression { params.ENVIRONMENT == 'testing' }
            }
            steps {
                bat 'mvn test'
            }
        }

        stage('Maven Build') {
            steps {
                bat 'mvn clean install -DskipTests=true'
            }
        }

        stage('Docker Build') {
            steps {
                bat "docker build -t %DOCKER_IMAGE%:%BUILD_NUMBER% ."
            }
        }

        stage('Remove Old Container') {
            steps {
                bat "docker rm -f %CONTAINER_NAME% || exit 0"
            }
        }

        stage('Run New Container') {
            steps {
                // Map host port 8091 to container port 8080 to avoid conflicts with Jenkins
                bat "docker run -d -p 8091:8080 --name=%CONTAINER_NAME% %DOCKER_IMAGE%:%BUILD_NUMBER%"
            }
        }
    }

    post {
        success {
            echo "Pipeline completed successfully. Application is running at http://localhost:8091"
        }
        failure {
            echo "Pipeline failed. Check logs for details."
        }
    }
}
